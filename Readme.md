## couponable


## Usage

```js
import {
  discountable,
  totalDueNow,
  totalRecurring,
  totalLineOne,
  totalLineTwo
} from 'couponable';

discountable(amountInCents, percentOff, amountOffInCents);
totalDueNow(orderItem);
totalRecurring(orderItem);
totalLineOne(orderItem);
totalLineTwo(orderItem);
```
